## 6. IMPLEMENTACIÃ“N TÃ‰CNICA

### Composable para EstadÃ­sticas

```typescript
// composables/useStatistics.ts
import { computed } from 'vue'
import { useClippingsStore } from '@/stores/clippings'
import { startOfWeek, startOfMonth, format } from 'date-fns'

export function useStatistics() {
  const clippingsStore = useClippingsStore()
  
  // Basic metrics
  const totalClippings = computed(() => 
    clippingsStore.clippings.length
  )
  
  const totalBooks = computed(() => 
    new Set(clippingsStore.clippings.map(c => c.book)).size
  )
  
  const totalAuthors = computed(() => 
    new Set(clippingsStore.clippings.map(c => c.author)).size
  )
  
  // Type distribution
  const typeDistribution = computed(() => {
    const types = clippingsStore.clippings.reduce((acc, c) => {
      acc[c.type] = (acc[c.type] || 0) + 1
      return acc
    }, {} as Record<string, number>)
    
    return [
      { type: 'Highlights', count: types.highlight || 0 },
      { type: 'Notes', count: types.note || 0 },
      { type: 'Bookmarks', count: types.bookmark || 0 }
    ]
  })
  
  // Top books
  const topBooks = computed(() => {
    const bookCounts = clippingsStore.clippings.reduce((acc, c) => {
      const key = `${c.book}::${c.author}`
      acc[key] = (acc[key] || 0) + 1
      return acc
    }, {} as Record<string, number>)
    
    return Object.entries(bookCounts)
      .map(([key, count]) => {
        const [book, author] = key.split('::')
        return { book, author, count }
      })
      .sort((a, b) => b.count - a.count)
      .slice(0, 10)
  })
  
  // Timeline data
  const timelineData = computed(() => {
    const grouped = clippingsStore.clippings.reduce((acc, c) => {
      const month = format(c.date, 'yyyy-MM')
      acc[month] = (acc[month] || 0) + 1
      return acc
    }, {} as Record<string, number>)
    
    return Object.entries(grouped)
      .map(([month, count]) => ({ month, count }))
      .sort((a, b) => a.month.localeCompare(b.month))
  })
  
  // Heatmap data
  const heatmapData = computed(() => {
    const data = Array.from({ length: 7 }, (_, day) =>
      Array.from({ length: 6 }, (_, hour) => ({
        day,
        hour: hour * 4 + 6, // 6, 10, 14, 18, 22, 2
        count: 0
      }))
    ).flat()
    
    clippingsStore.clippings.forEach(c => {
      const day = c.date.getDay()
      const hour = c.date.getHours()
      const slot = Math.floor((hour - 6) / 4)
      const cell = data.find(d => d.day === day && d.hour === slot * 4 + 6)
      if (cell) cell.count++
    })
    
    return data
  })
  
  return {
    totalClippings,
    totalBooks,
    totalAuthors,
    typeDistribution,
    topBooks,
    timelineData,
    heatmapData
  }
}
```
